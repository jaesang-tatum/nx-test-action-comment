import { Injectable } from '@nestjs/common';
import { CreateAwsLogBodyDto } from '../router/cloud-log/dto';
import { CommonLogSchemaType } from './cloudLog.type';
import { AbstractCloudLogMapper } from './cloudLogMapper';
 
@Injectable()
export class AwsLogMapper extends AbstractCloudLogMapper {
    protected execMappingAlg(cloudLog: CreateAwsLogBodyDto): CommonLogSchemaType[] {
        const commonPart = {
            targetId: cloudLog.targetId,
            tenantId: cloudLog.tenantId,
            provider: cloudLog.provider,
            eventId: cloudLog.eventID,
            eventName: cloudLog.eventName,
            eventType: cloudLog.eventType,
            origin: cloudLog,
            ipAddr: cloudLog.sourceIPAddress ?? null,
            region: cloudLog.awsRegion,
            identityId: cloudLog.userIdentity?.accessKeyId ?? null,
            createdAt: new Date(cloudLog['@timestamp']),
        };
 
        const resourceList = cloudLog.resources;
        if (resourceList === undefined || resourceList.length === 0) {
            return [{ ...commonPart, resourceId: null, resourceName: null, resourceType: null }];
        }
        return resourceList.map((resource) => {
            return {
                ...commonPart,
                resourceId: resource.arn ?? null,
                resourceName: null,
                resourceType: resource.type ?? null,
            };
        });
    }
}
