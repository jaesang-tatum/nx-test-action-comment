import CryptoJS from 'crypto-js';
import { Request } from '@backend-x/common/request';
import { NcpSubProviderType } from '@backend-x/common/type/provider.type';
import { AbstractCloudProviderStrategy } from '../cloudProviderStrategy';
import { NcpGetActivityListReturnType } from './type';
import { NcpGetActivityListRequestFailureException } from './exeption';
import { ManualPipelineConfig } from 'product/log-collector/src/collector';
import { NcpLogSchema } from 'product/log-collector/src/log-schema';
import dayjs from 'dayjs';
 
export class NcpStrategy extends AbstractCloudProviderStrategy {
    private readonly fromRequester: Request;
 
    public static readonly PathOfGetActivityListApi = '/api/v1/activities';
    public static readonly NcpUrl: Record<NcpSubProviderType, string> = {
        NCP: 'https://cloudactivitytracer.apigw.ntruss.com',
        NCP_FIN: 'https://cloudactivitytracer.apigw.fin-ntruss.com',
        NCP_GOV: 'https://cloudactivitytracer.apigw.gov-ntruss.com',
    };
 
    constructor(
        private readonly ncpCategory: NcpSubProviderType,
        private readonly toRequester: Request,
        pipelineConfig: ManualPipelineConfig<'NCP'>
    ) {
        super(pipelineConfig);
        this.fromRequester = new Request(NcpStrategy.NcpUrl[ncpCategory]);
    }
 
    public async collectCloudLog(tenantId: string, timesecondRangeForCollect: number): Promise<void> {
        let isNext = false;
        let pageIndex = 0;
        let resultList: NcpLogSchema[] = [];
 
        do {
            const result = await this.requestGetActivityListApi(timesecondRangeForCollect, pageIndex);
            resultList = [...resultList, ...result.items];
            isNext = result.hasMore;
            pageIndex += 1;
        } while (isNext);
 
        await this.sendCloudLogToLogSink(tenantId, resultList);
    }
 
    private async sendCloudLogToLogSink(tenantId: string, resultList: NcpLogSchema[]): Promise<void> {
        const { targetId } = this.pipelineConfig.getConfig();
        const PromiseToSendLogList = resultList.map((ncpLog: NcpLogSchema) => {
            return this.toRequester.send('/cloud-log/ncp', {
                method: 'post',
                data: { ...ncpLog, targetId, tenantId, provider: this.ncpCategory },
            });
        });
        Promise.all(PromiseToSendLogList);
    }
 
    private createSignature(now: number) {
        const { accessKey, secretKey } = this.pipelineConfig.getConfig();
        return CryptoJS.HmacSHA256(
            `POST ${NcpStrategy.PathOfGetActivityListApi}\n${now.toString()}\n${accessKey}`,
            secretKey
        ).toString(CryptoJS.enc.Base64);
    }
 
    private async requestGetActivityListApi(
        timesecondRangeForCollect: number,
        pageIndex: number
    ): Promise<NcpGetActivityListReturnType> {
        const { targetId, accessKey, nrn } = this.pipelineConfig.getConfig();
        const now = dayjs().valueOf();
        const result = await this.fromRequester.send<NcpGetActivityListReturnType>(
            NcpStrategy.PathOfGetActivityListApi,
            {
                headers: {
                    'Content-Type': 'application/json; charset=utf-8',
                    'x-ncp-apigw-timestamp': now.toString(),
                    'x-ncp-iam-access-key': accessKey,
                    'x-ncp-apigw-signature-v2': this.createSignature(now),
                },
                method: 'post',
                data: {
                    fromEventTime: now - timesecondRangeForCollect * 1000,
                    toEventTime: now,
                    pageIndex,
                    nrn,
                },
            }
        );
 
        if (result === undefined) {
            throw new NcpGetActivityListRequestFailureException(targetId);
        }
        return result;
    }
}
