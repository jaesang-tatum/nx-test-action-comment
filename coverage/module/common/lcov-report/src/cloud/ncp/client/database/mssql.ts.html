import {
    CloudMssqlInstanceDetailResponseType,
    CloudMssqlInstanceListResponseType,
    CloudMssqlInstanceType,
} from '@backend-x/common/type/cloud/ncp/resource/database/mssql.type';
import { NcpCollector } from '../../ncpCollector';
 
export class Mssql extends NcpCollector {
    public async get(cloudMssqlInstanceNo: string, regionCode: string): Promise<CloudMssqlInstanceType | undefined> {
        const url = `https://${this.client.subDomain}.apigw.${this.client.apiDomain}.com/vmssql/v2/getCloudMssqlInstanceDetail?pageSize=5000&cloudMssqlInstanceNo=${cloudMssqlInstanceNo}&responseFormatType=json&regionCode=${regionCode}`;
        const res: CloudMssqlInstanceDetailResponseType = await this.client.get(url, this.key);
        return res.getCloudMssqlInstanceDetailResponse?.cloudMssqlInstanceList?.[0];
    }
    public async list(regionCode: string): Promise<CloudMssqlInstanceType[]> {
        const url = `https://${this.client.subDomain}.apigw.${this.client.apiDomain}.com/vmssql/v2/getCloudMssqlInstanceList?pageSize=5000&responseFormatType=json&regionCode=${regionCode}`;
        const res: CloudMssqlInstanceListResponseType = await this.client.get(url, this.key);
        return res.getCloudMssqlInstanceListResponse?.cloudMssqlInstanceList ?? [];
    }
}
